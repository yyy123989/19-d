# STM32F407VET6 TFT + AD7606B Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a CubeMX/HAL firmware project that drives the board-reserved FSMC TFT interface and reads AD7606B samples from a GPIOB 16-bit parallel bus.

**Architecture:** CubeMX owns clock, GPIO, FSMC, NVIC, and optional UART initialization. Firmware is split into a board pin layer, TFT low-level FSMC bus functions, LCD controller initialization/drawing functions, and AD7606B control/read functions. The first integration screen displays raw AD7606B values on the TFT.

**Tech Stack:** STM32CubeMX, STM32CubeIDE or Makefile-generated STM32Cube project, STM32 HAL, STM32F407VET6, FSMC 8080 TFT, AD7606B parallel interface.

---

## File Structure

- Create or generate: `TFT_AD7606B.ioc` for CubeMX configuration.
- Modify: `Core/Inc/main.h` for GPIO labels generated by CubeMX.
- Modify: `Core/Src/main.c` for startup sequence and first integration loop.
- Create: `Core/Inc/board_pins.h` for stable project-level pin naming notes when CubeMX labels are incomplete.
- Create: `Core/Inc/tft_fsmc.h` and `Core/Src/tft_fsmc.c` for LCD bus writes and reads.
- Create: `Core/Inc/lcd.h` and `Core/Src/lcd.c` for LCD controller initialization and primitive drawing.
- Create: `Core/Inc/ad7606b.h` and `Core/Src/ad7606b.c` for AD7606B reset, conversion, busy wait, and parallel data reads.

- [ ] **Step 2: Create LCD source**

Create `Core/Src/lcd.c` using the actual LCD controller command set. For an ILI9341-compatible screen, start with:

```c
#include "lcd.h"
#include "tft_fsmc.h"
#include "main.h"

#define LCD_WIDTH   240U
#define LCD_HEIGHT  320U

static void LCD_WriteReg(uint16_t reg, uint16_t value)
{
    TFT_WriteCommand(reg);
    TFT_WriteData(value);
}

static void LCD_SetWindow(uint16_t x0, uint16_t y0, uint16_t x1, uint16_t y1)
{
    TFT_WriteCommand(0x2A);
    TFT_WriteData(x0 >> 8);
    TFT_WriteData(x0 & 0xFF);
    TFT_WriteData(x1 >> 8);
    TFT_WriteData(x1 & 0xFF);

    TFT_WriteCommand(0x2B);
    TFT_WriteData(y0 >> 8);
    TFT_WriteData(y0 & 0xFF);
    TFT_WriteData(y1 >> 8);
    TFT_WriteData(y1 & 0xFF);

    TFT_WriteCommand(0x2C);
}

void LCD_Init(void)
{
    HAL_Delay(50);
    TFT_WriteCommand(0x01);
    HAL_Delay(120);

    TFT_WriteCommand(0x11);
    HAL_Delay(120);

    LCD_WriteReg(0x3A, 0x55);
    TFT_WriteCommand(0x29);
}

void LCD_Fill(uint16_t color)
{
    uint32_t i;

    LCD_SetWindow(0, 0, LCD_WIDTH - 1U, LCD_HEIGHT - 1U);
    for (i = 0; i < (LCD_WIDTH * LCD_HEIGHT); i++) {
        TFT_WriteData(color);
    }
}

void LCD_DrawPixel(uint16_t x, uint16_t y, uint16_t color)
{
    if ((x >= LCD_WIDTH) || (y >= LCD_HEIGHT)) {
        return;
    }

    LCD_SetWindow(x, y, x, y);
    TFT_WriteData(color);
}
```

#endif

- [ ] **Step 2: Create AD7606B source**

Create `Core/Src/ad7606b.c`:

```c
#include "ad7606b.h"
#include "main.h"

static uint16_t AD7606B_ReadWord(void)
{
    HAL_GPIO_WritePin(AD_CS_GPIO_Port, AD_CS_Pin, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(AD_RD_GPIO_Port, AD_RD_Pin, GPIO_PIN_RESET);
    __NOP();
    __NOP();
    uint16_t value = (uint16_t)(GPIOB->IDR & 0xFFFFU);
    HAL_GPIO_WritePin(AD_RD_GPIO_Port, AD_RD_Pin, GPIO_PIN_SET);
    HAL_GPIO_WritePin(AD_CS_GPIO_Port, AD_CS_Pin, GPIO_PIN_SET);
    return value;
}

void AD7606B_Init(void)
{
    HAL_GPIO_WritePin(AD_CS_GPIO_Port, AD_CS_Pin, GPIO_PIN_SET);
    HAL_GPIO_WritePin(AD_RD_GPIO_Port, AD_RD_Pin, GPIO_PIN_SET);
    HAL_GPIO_WritePin(AD_WR_GPIO_Port, AD_WR_Pin, GPIO_PIN_SET);
    HAL_GPIO_WritePin(AD_CONVST_GPIO_Port, AD_CONVST_Pin, GPIO_PIN_SET);
    HAL_GPIO_WritePin(AD_STBY_GPIO_Port, AD_STBY_Pin, GPIO_PIN_SET);
    HAL_GPIO_WritePin(AD_PAR_SER_GPIO_Port, AD_PAR_SER_Pin, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(AD_RANGE_GPIO_Port, AD_RANGE_Pin, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(AD_OS0_GPIO_Port, AD_OS0_Pin, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(AD_OS1_GPIO_Port, AD_OS1_Pin, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(AD_OS2_GPIO_Port, AD_OS2_Pin, GPIO_PIN_RESET);

    HAL_GPIO_WritePin(AD_RESET_GPIO_Port, AD_RESET_Pin, GPIO_PIN_SET);
    HAL_Delay(1);
    HAL_GPIO_WritePin(AD_RESET_GPIO_Port, AD_RESET_Pin, GPIO_PIN_RESET);
    HAL_Delay(1);
}

void AD7606B_StartConversion(void)
{
    HAL_GPIO_WritePin(AD_CONVST_GPIO_Port, AD_CONVST_Pin, GPIO_PIN_RESET);
    __NOP();
    __NOP();
    HAL_GPIO_WritePin(AD_CONVST_GPIO_Port, AD_CONVST_Pin, GPIO_PIN_SET);
}

uint8_t AD7606B_ReadChannels(int16_t samples[AD7606B_CHANNEL_COUNT])
{
    uint32_t timeout = 1000000U;

    while ((HAL_GPIO_ReadPin(AD_BUSY_GPIO_Port, AD_BUSY_Pin) == GPIO_PIN_SET) && (timeout > 0U)) {
        timeout--;
    }

    if (timeout == 0U) {
        return 0U;
    }

    for (uint8_t i = 0; i < AD7606B_CHANNEL_COUNT; i++) {
        samples[i] = (int16_t)AD7606B_ReadWord();
    }

    return 1U;
}
```

```c
#define TFT_RAM_ADDR  ((volatile uint16_t *)0x60080000U)
```

For `A18` on a 16-bit FSMC bus, keep `0x60080000`. For another address line, use the matching FSMC address offset.
